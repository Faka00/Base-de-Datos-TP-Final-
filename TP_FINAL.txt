USE (Nombre Base de datos) /* Ingresar Nombre de Base de datos */

/* Tablas principales sin relaciones */

CREATE TABLE pais (
    id_pais     INT          IDENTITY(1,1) NOT NULL,
    nombre_pais NVARCHAR(100) NOT NULL,
    codigo_iso  CHAR(3)       NOT NULL,
    CONSTRAINT PK_pais PRIMARY KEY (id_pais)
);

CREATE TABLE rol_empleado (
    id_rol     INT          IDENTITY(1,1) NOT NULL,
    nombre_rol NVARCHAR(50)  NOT NULL,
    CONSTRAINT PK_rol_empleado PRIMARY KEY (id_rol)
);

CREATE TABLE tipo_habitacion (
    id_tipo_habitacion INT          IDENTITY(1,1) NOT NULL,
    nombre_tipo        NVARCHAR(50)  NOT NULL,
    capacidad_max      INT           NOT NULL,
    descripcion        NVARCHAR(255) NULL,
    CONSTRAINT PK_tipo_habitacion PRIMARY KEY (id_tipo_habitacion)
);

CREATE TABLE comodidad (
    id_comodidad     INT          IDENTITY(1,1) NOT NULL,
    nombre_comodidad NVARCHAR(80)  NOT NULL,
    CONSTRAINT PK_comodidad PRIMARY KEY (id_comodidad)
);

CREATE TABLE estado_reserva (
    id_estado_reserva INT          IDENTITY(1,1) NOT NULL,
    descripcion       NVARCHAR(50)  NOT NULL,
    CONSTRAINT PK_estado_reserva PRIMARY KEY (id_estado_reserva)
);

CREATE TABLE servicio_extra (
    id_servicio INT           IDENTITY(1,1) NOT NULL,
    nombre      NVARCHAR(100)  NOT NULL,
    precio      DECIMAL(10,2)  NOT NULL,
    CONSTRAINT PK_servicio_extra PRIMARY KEY (id_servicio)
);

CREATE TABLE metodo_pago (
    id_metodo_pago INT          IDENTITY(1,1) NOT NULL,
    descripcion    NVARCHAR(50)  NOT NULL,
    CONSTRAINT PK_metodo_pago PRIMARY KEY (id_metodo_pago)
);

/* ---------- Hotel y dependientes ---------- */

CREATE TABLE hotel (
    id_hotel  INT           IDENTITY(1,1) NOT NULL,
    nombre    NVARCHAR(120)  NOT NULL,
    direccion NVARCHAR(180)  NOT NULL,
    ciudad    NVARCHAR(100)  NOT NULL,
    id_pais   INT            NOT NULL,
    telefono  NVARCHAR(30)   NULL,
    CONSTRAINT PK_hotel PRIMARY KEY (id_hotel),
    CONSTRAINT FK_hotel_pais FOREIGN KEY (id_pais) REFERENCES pais (id_pais)
);

CREATE TABLE empleado (
    id_empleado       INT           IDENTITY(1,1) NOT NULL,
    id_rol            INT            NOT NULL,
    id_hotel          INT            NOT NULL,
    nombre            NVARCHAR(80)   NOT NULL,
    apellido          NVARCHAR(80)   NOT NULL,
    email_corporativo NVARCHAR(150)  NULL,
    activo            BIT            NOT NULL,
    CONSTRAINT PK_empleado PRIMARY KEY (id_empleado),
    CONSTRAINT FK_empleado_rol   FOREIGN KEY (id_rol)   REFERENCES rol_empleado (id_rol),
    CONSTRAINT FK_empleado_hotel FOREIGN KEY (id_hotel) REFERENCES hotel (id_hotel)
);

CREATE TABLE habitacion (
    id_habitacion      INT           IDENTITY(1,1) NOT NULL,
    id_hotel           INT            NOT NULL,
    id_tipo_habitacion INT            NOT NULL,
    numero_habitacion  NVARCHAR(10)   NOT NULL,
    piso               INT            NULL,
    precio_base_noche  DECIMAL(10,2)  NOT NULL,
    CONSTRAINT PK_habitacion PRIMARY KEY (id_habitacion),
    CONSTRAINT FK_habitacion_hotel FOREIGN KEY (id_hotel)           REFERENCES hotel (id_hotel),
    CONSTRAINT FK_habitacion_tipo  FOREIGN KEY (id_tipo_habitacion) REFERENCES tipo_habitacion (id_tipo_habitacion)
);

/* Tipo_habitacion <-> comodidad */
CREATE TABLE habitacion_comodidad (
    id_tipo_habitacion INT NOT NULL,
    id_comodidad       INT NOT NULL,
    CONSTRAINT PK_habitacion_comodidad PRIMARY KEY (id_tipo_habitacion, id_comodidad),
    CONSTRAINT FK_hc_tipo      FOREIGN KEY (id_tipo_habitacion) REFERENCES tipo_habitacion (id_tipo_habitacion),
    CONSTRAINT FK_hc_comodidad FOREIGN KEY (id_comodidad)       REFERENCES comodidad (id_comodidad)
);

/* ---------- Huespedes ---------- */

CREATE TABLE huesped (
    id_huesped       INT           IDENTITY(1,1) NOT NULL,
    id_pais_origen   INT            NULL,
    nombre           NVARCHAR(80)   NOT NULL,
    apellido         NVARCHAR(80)   NOT NULL,
    email            NVARCHAR(150)  NULL,
    telefono         NVARCHAR(30)   NULL,
    pasaporte_dni    NVARCHAR(40)   NOT NULL,
    fecha_nacimiento DATE           NULL,
    CONSTRAINT PK_huesped PRIMARY KEY (id_huesped),
    CONSTRAINT FK_huesped_pais FOREIGN KEY (id_pais_origen) REFERENCES pais (id_pais)
);

CREATE TABLE programa_fidelidad (
    id_fidelidad      INT  IDENTITY(1,1) NOT NULL,
    id_huesped        INT  NOT NULL,
    puntos_acumulados INT  NOT NULL,
    nivel             NVARCHAR(30) NULL,
    fecha_alta        DATE NOT NULL,
    CONSTRAINT PK_programa_fidelidad PRIMARY KEY (id_fidelidad),
    CONSTRAINT FK_fidelidad_huesped FOREIGN KEY (id_huesped) REFERENCES huesped (id_huesped)
);

/* ---------- Reservas ---------- */

CREATE TABLE reserva (
    id_reserva          INT      IDENTITY(1,1) NOT NULL,
    id_huesped          INT      NOT NULL,
    id_estado_reserva   INT      NOT NULL,
    fecha_reserva       DATETIME2 NOT NULL,
    fecha_checkin_prev  DATE     NOT NULL,
    fecha_checkout_prev DATE     NOT NULL,
    CONSTRAINT PK_reserva PRIMARY KEY (id_reserva),
    CONSTRAINT FK_reserva_huesped FOREIGN KEY (id_huesped)        REFERENCES huesped (id_huesped),
    CONSTRAINT FK_reserva_estado  FOREIGN KEY (id_estado_reserva) REFERENCES estado_reserva (id_estado_reserva)
);

CREATE TABLE detalle_reserva (
    id_detalle_reserva INT           IDENTITY(1,1) NOT NULL,
    id_habitacion      INT            NOT NULL,
    id_reserva         INT            NOT NULL,
    fecha_inicio       DATE           NOT NULL,
    fecha_fin          DATE           NOT NULL,
    precio_noche       DECIMAL(10,2)  NOT NULL,
    CONSTRAINT PK_detalle_reserva PRIMARY KEY (id_detalle_reserva),
    CONSTRAINT FK_det_habitacion FOREIGN KEY (id_habitacion) REFERENCES habitacion (id_habitacion),
    CONSTRAINT FK_det_reserva    FOREIGN KEY (id_reserva)    REFERENCES reserva (id_reserva)
);

CREATE TABLE registro_estadia (
    id_registro         INT       IDENTITY(1,1) NOT NULL,
    id_detalle_reserva  INT       NOT NULL,
    fecha_hora_checkin  DATETIME2 NULL,
    fecha_hora_checkout DATETIME2 NULL,
    CONSTRAINT PK_registro_estadia PRIMARY KEY (id_registro),
    CONSTRAINT FK_registro_detalle FOREIGN KEY (id_detalle_reserva) REFERENCES detalle_reserva (id_detalle_reserva)
);

/* Reserva <-> servicio_extra */
CREATE TABLE reserva_servicio (
    id_reserva  INT NOT NULL,
    id_servicio INT NOT NULL,
    cantidad    INT NOT NULL,
    CONSTRAINT PK_reserva_servicio PRIMARY KEY (id_reserva, id_servicio),
    CONSTRAINT FK_rs_reserva  FOREIGN KEY (id_reserva)  REFERENCES reserva (id_reserva),
    CONSTRAINT FK_rs_servicio FOREIGN KEY (id_servicio) REFERENCES servicio_extra (id_servicio)
);

/* ---------- Eventos ---------- */

CREATE TABLE evento (
    id_evento         INT           IDENTITY(1,1) NOT NULL,
    id_hotel          INT            NOT NULL,
    nombre_evento     NVARCHAR(120)  NOT NULL,
    fecha_hora        DATETIME2      NOT NULL,
    costo_inscripcion DECIMAL(10,2)  NOT NULL,
    cupo_maximo       INT            NULL,
    CONSTRAINT PK_evento PRIMARY KEY (id_evento),
    CONSTRAINT FK_evento_hotel FOREIGN KEY (id_hotel) REFERENCES hotel (id_hotel)
);

CREATE TABLE inscripcion_evento (
    id_inscripcion    INT       IDENTITY(1,1) NOT NULL,
    id_evento         INT       NOT NULL,
    id_huesped        INT       NOT NULL,
    fecha_inscripcion DATETIME2 NOT NULL,
    CONSTRAINT PK_inscripcion_evento PRIMARY KEY (id_inscripcion),
    CONSTRAINT FK_insc_evento  FOREIGN KEY (id_evento)  REFERENCES evento (id_evento),
    CONSTRAINT FK_insc_huesped FOREIGN KEY (id_huesped) REFERENCES huesped (id_huesped)
);

/* ---------- Coworking ---------- */

CREATE TABLE coworking (
    id_espacio            INT          IDENTITY(1,1) NOT NULL,
    id_hotel              INT           NOT NULL,
    nombre_area           NVARCHAR(100) NOT NULL,
    capacidad_escritorios INT           NOT NULL,
    CONSTRAINT PK_coworking PRIMARY KEY (id_espacio),
    CONSTRAINT FK_cowork_hotel FOREIGN KEY (id_hotel) REFERENCES hotel (id_hotel)
);

CREATE TABLE reserva_cowork (
    id_reserva_cowork INT           IDENTITY(1,1) NOT NULL,
    id_huesped        INT            NOT NULL,
    id_espacio        INT            NOT NULL,
    fecha_uso         DATE           NOT NULL,
    tipo_pase         NVARCHAR(30)   NULL,
    precio_pagado     DECIMAL(10,2)  NOT NULL,
    CONSTRAINT PK_reserva_cowork PRIMARY KEY (id_reserva_cowork),
    CONSTRAINT FK_rc_huesped FOREIGN KEY (id_huesped) REFERENCES huesped (id_huesped),
    CONSTRAINT FK_rc_espacio FOREIGN KEY (id_espacio) REFERENCES coworking (id_espacio)
);

/* ---------- Facturacion ---------- */

CREATE TABLE factura (
    id_factura    INT           IDENTITY(1,1) NOT NULL,
    id_reserva    INT            NOT NULL,
    fecha_emision DATETIME2      NOT NULL,
    monto_total   DECIMAL(12,2)  NOT NULL,
    impuesto      DECIMAL(12,2)  NOT NULL,
    CONSTRAINT PK_factura PRIMARY KEY (id_factura),
    CONSTRAINT FK_factura_reserva FOREIGN KEY (id_reserva) REFERENCES reserva (id_reserva)
);

CREATE TABLE pago (
    id_pago                 INT           IDENTITY(1,1) NOT NULL,
    id_factura              INT            NOT NULL,
    id_metodo_pago          INT            NOT NULL,
    fecha_pago              DATETIME2      NOT NULL,
    monto_pagado            DECIMAL(12,2)  NOT NULL,
    transaccion_pasarela_id NVARCHAR(100)  NULL,
    CONSTRAINT PK_pago PRIMARY KEY (id_pago),
    CONSTRAINT FK_pago_factura FOREIGN KEY (id_factura)     REFERENCES factura (id_factura),
    CONSTRAINT FK_pago_metodo  FOREIGN KEY (id_metodo_pago) REFERENCES metodo_pago (id_metodo_pago)
);

/* ---------- Mantenimiento ---------- */

CREATE TABLE orden_mantenimiento (
    id_orden_mant        INT           IDENTITY(1,1) NOT NULL,
    id_empleado_asignado INT            NULL,
    id_habitacion        INT            NOT NULL,
    fecha_solicitud      DATETIME2      NOT NULL,
    descripcion_problema NVARCHAR(255)  NOT NULL,
    estado_orden         NVARCHAR(30)   NOT NULL,
    CONSTRAINT PK_orden_mantenimiento PRIMARY KEY (id_orden_mant),
    CONSTRAINT FK_orden_empleado   FOREIGN KEY (id_empleado_asignado) REFERENCES empleado (id_empleado),
    CONSTRAINT FK_orden_habitacion FOREIGN KEY (id_habitacion)        REFERENCES habitacion (id_habitacion)
);
